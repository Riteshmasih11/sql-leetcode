with cte1 as (
    select max(customer_id) as max_id
    from customers
), cte2 as (
    select 1 as ids -- anchor member
    union all   
    select ids + 1 -- recursive member
    from cte2
    where ids < (select max_id from cte1)
)

select ids
from cte2
where ids not in (select customer_id from customers)
